	text = {
		trigger = {
			set_temp_variable = { NOV_RAPP_Display_Total = NOV_RAPP_lower_seats }
			add_to_temp_variable = { NOV_RAPP_Display_Total = NOV_RAPP_upper_seats }

	}
	text = {
		localization_key = "[?NOV_RAPP_Display_Total|Y0]"
	}
}
