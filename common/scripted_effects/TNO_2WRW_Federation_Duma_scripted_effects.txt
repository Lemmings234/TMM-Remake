
# Interaction Cooldown

NOV_Interaction_cooldown = {
	custom_effect_tooltip = ERS_Interaction_cooldown_tt
	hidden_effect = {
		set_country_flag = NOV_Presidential_Interaction
		country_event = { id = NOV_federalassembly.0 days = 14 }
	}
}

# Presidential Support Effects

NOV_Increased_Presidental_Support_Low = {
	custom_effect_tooltip =  NOV_Increased_Presidental_Support_Low_tt
	hidden_effect = {
		if = {
			limit = {
				has_government = conservatism
			}
			add_popularity = {
				ideology = conservatism
				popularity = 0.035
			}
		}
		else_if = {
			limit = {
				has_government = liberalism
			}
			add_popularity = {
				ideology = liberalism
				popularity = 0.035
			}
		}
		else_if = {
			limit = {
				has_government = progressivism
			}
			add_popularity = {
				ideology = progressivism
				popularity = 0.035
			}
		}
		else = {
			add_popularity = {
				ideology = paternalism
				popularity = 0.035
			}
		}
	}
}

# State Duma

NOV_Increase_Duma_Seats_Low = {
	custom_effect_tooltip = NOV_Increase_Duma_Seats_Low_tt
	hidden_effect = {
		subtract_from_variable = {
			var = NOV_SIL_lower_seats
			value = 2
		}
		add_to_variable = {
			var = NOV_RAPP_lower_seats
			value = 2
		}
		NOV_Get_Lower_House_Array = yes
