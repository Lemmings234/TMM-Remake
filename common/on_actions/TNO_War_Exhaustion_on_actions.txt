on_actions = {

	### war exhaustion gain from casualties ##

	on_daily = {
		effect = {
			if = {
				limit = {
					ROOT = {
						is_russian_nation = yes
						has_war_with = GER
						has_country_flag = RUS_2WRW_ongoing
						check_variable = {
							var = casualties_k
							value = RUS_casualty_tracker
							compare = greater_than_or_equals
						}
					}
				}
				set_temp_variable = { temp_exhaustion = 0.75 }
				WRW_add_war_exhaustion = yes
				add_to_variable = { RUS_casualty_tracker = 50 }
			}
			if = {
				limit = {
					ROOT = {
						original_tag = GER
						has_country_flag = RUS_2WRW_ongoing
						check_variable = {
							var = casualties_k
							value = GER_casualty_tracker
							compare = greater_than_or_equals
						}
					}
				}
				set_temp_variable = { temp_exhaustion = 0.55 }
				WRW_add_war_exhaustion = yes
				add_to_variable = { GER_casualty_tracker = 50 }
			}
		}
	}
	
	### monthly war exhaustion tick ###
	
	on_monthly = {
		effect = {
			if = {
				limit = {
					ROOT = {
						has_country_flag = RUS_2WRW_ongoing
					}
				}
				set_temp_variable = { temp_exhaustion = 5 }
				WRW_add_war_exhaustion = yes
			}
		}
	}
}
