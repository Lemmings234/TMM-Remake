	TNO_2WRW_Global_Conflicts_Russia = {
		window_name = "TNO_2WRW_Global_Conflicts_Russia"
		context_type = decision_category
		
		visible = {
			OR = {
				check_variable = { 2WRW_global_conflicts = 1 }
				check_variable = { 2WRW_global_conflicts = 2 }
			}
		}
		
		effects = {
			conflict_icon2_click = {
				# Pre-2WRW Proxy Wars
				if = {
					limit = { check_variable = { 2WRW_global_conflicts = 1 } }
					set_variable = { 2WRW_global_conflicts = 2 }
				}
				else_if = {
					limit = { check_variable = { 2WRW_global_conflicts = 2 } }
					set_variable = { 2WRW_global_conflicts = 1 }
				}
			}
		}
		
		triggers = {
			tab2_visible = {
				AND = {
					has_global_flag = Iranian_Civil_War

				}
			}
			conflict_icon2_visible = {
				AND = {
					has_global_flag = Iranian_Civil_War
					has_global_flag = TNO_2WRW_Second_Great_Northern_War
				}
			}
		}
		
		properties = {

				image = "[This.GetTabPicture]"

			}
		}
	}
