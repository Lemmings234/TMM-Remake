scripted_gui = {
	RUS_Siberian_Railway_GUI_Opener = {
		context_type = decision_category
		window_name = "RUS_Siberian_Railway_GUI_Opener"

		visible = {
			tag = RUS
			NOT = {
				has_global_flag = AFTER_MIDNIGHT_COMMENCED
			}
		}

		effects = {
			RUS_Open_Siberian_Railway_Button_click = {
				if = {
					limit = {
						has_country_flag = RUS_Trans_Siberian_GUI_visible_flag 
					}
					clr_country_flag = RUS_Trans_Siberian_GUI_visible_flag
				}
				else = { set_country_flag = RUS_Trans_Siberian_GUI_visible_flag }
			}
		}
	}
	
	RUS_Siberian_Railway_GUI = {
		context_type = player_context
		window_name = "RUS_Siberian_Railway_GUI"

		visible = {
			has_country_flag = RUS_Trans_Siberian_GUI_visible_flag
			NOT = {
				has_global_flag = AFTER_MIDNIGHT_COMMENCED
			}
		}
		effects = {
			CentralSiberiaStateSelectedButton_click = {
				if = {
					limit = {
						has_country_flag = RUS_Trans_Siberian_Central_Siberia_Selected_flag
					}
					clr_country_flag = RUS_Trans_Siberian_Central_Siberia_Selected_flag
				}
				else = { set_country_flag = RUS_Trans_Siberian_Central_Siberia_Selected_flag }
			}
			RUS_Siberian_Exit_close_button_click = {
				clr_country_flag = RUS_Trans_Siberian_GUI_visible_flag
				if = {
					limit = {
						has_country_flag = RUS_Trans_Siberian_Central_Siberia_Selected_flag
					}
					clr_country_flag = RUS_Trans_Siberian_Central_Siberia_Selected_flag
				}
			}
		}
	}
	
	RUS_siberianrailway_centralsibiera_info = {

		}
	}
