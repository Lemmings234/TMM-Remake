news_event = { #Star War, 1977
	id = star_wars.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event star_wars.1"}
	title = star_wars.1.t
	desc = star_wars.1.desc
	picture = GFX_star_wars_news_event
	
	is_triggered_only = yes

	option = {
		name = star_wars.1.a
		hidden_effect = {
			JAP = {country_event = { id = star_wars.2 days = 7 } }
			#3 triggered from japan event
			every_country = { #Einhenspakt
				limit = {
					is_in_faction_with = GER
					NOT = { tag = ENG }
				}
				country_event = {id = star_wars.4 days = 7}
			}
			ENG = {
				if = {
					limit = { has_global_flag = ENG_GF_WonECW }
					country_event = {id = star_wars.5 days = 7 }
				}
				else = {
					country_event = {id = star_wars.6 days = 7 }
				}
			}
			USA = {country_event = {id = star_wars.7 days = 7} }
			RUS = {
				if = {
					limit = {
						NOT = {
							has_government = national_socialism
							has_government = ultranationalism
							has_government = esoteric_nazism
						}
					}
				country_event = { id = star_wars.8 days = 14 }
				}
			}
		}
	}
}
