country_event = { #Just Moscowien
	id = RUS_victory.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event RUS_victory.1"}
	title = RUS_victory.1.t
	desc = RUS_victory.1.desc
	picture = GFX_report_event_russia_red_army_poster
	
	is_triggered_only = yes
	
	option = {
		name = RUS_victory.1.a
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				OR = {
					has_game_rule = {
						rule = RUS_SWRW_PEACEDEAL
						option = RUS_SWRW_PEACEDEAL_CAUCASUS
					}
					has_game_rule = {
						rule = RUS_SWRW_PEACEDEAL
						option = RUS_SWRW_PEACEDEAL_UKRAINE
					}
					has_game_rule = {
						rule = RUS_SWRW_PEACEDEAL
						option = RUS_SWRW_PEACEDEAL_FULL_VICTORY
					}
					has_game_rule = {
						rule = RUS_SWRW_PEACEDEAL
						option = RUS_SWRW_PEACEDEAL_TOTAL_DEFEAT
					}
				}
			}
		}
		if = {
			limit = {
				AND = {
					original_tag = NOV
					has_global_flag = RUS_shukshin_unification
				}
			}
			RUS_moskowien_shukshin_occupation = yes
			RUS_shukshin_petrograd_occupation = yes
			hidden_effect = {
				add_ideas = RUS_reconstruction_era
				set_global_flag = RUS_no_caucasus
				set_country_flag = NOV_peace_deal_1
				set_global_flag = RUS_2WRW_Victory
				set_global_flag = RUS_2WRW_End
				clr_country_flag = RUS_2WRW_ongoing
				load_focus_tree = {	tree = NOV_novosibirsk_victory_shukshin keep_completed = no }
			}
		}
		else = {
			RUS_moskowien_annexation = yes
			set_global_flag = RUS_2WRW_Victory
			set_global_flag = RUS_2WRW_End
		}
		RUS = {
			set_temp_variable = { Victory_ID = 97 }
			set_temp_variable = { Victory_Outcome = 4 }
			set_temp_variable = { Victory_Points = 200 }
			Cold_War_GUI_ManipulateVictory = yes
			remove_dynamic_modifier = { modifier = RUS_war_exhaustion } 
	    	remove_dynamic_modifier = { modifier = WRW_tension } 
			RUS_remove_wargoals = yes
			clr_country_flag = RUS_2WRW_ongoing
		}
		GER = {
			set_temp_variable = { Victory_ID = 97 }
			set_temp_variable = { Victory_Outcome = 2 }
			set_temp_variable = { Victory_Points = -200 }
			Cold_War_GUI_ManipulateVictory = yes
			remove_dynamic_modifier = { modifier = GER_war_exhaustion } 
	    	remove_dynamic_modifier = { modifier = WRW_tension } 
			clr_country_flag = RUS_2WRW_ongoing
		}
		clr_global_flag = RUS_2WRW
	}
	
	option = {
		name = RUS_victory.1.b
		ai_chance = {
			factor = 80
			modifier = {
				factor = 0
				has_game_rule = { rule = RUS_SWRW_PEACEDEAL option = RUS_SWRW_PEACEDEAL_MOSCOW }
			}
		}
		set_temp_variable = { temp_exhaustion = 1 } 
		WRW_add_war_exhaustion = yes
	}
}
