add_namespace = RUS_fourthpower

country_event = { # Mountain War - Peace Deal One
	id = RUS_fourthpower.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event RUS_fourthpower.1" }
	title = RUS_fourthpower.1.t
	desc = RUS_fourthpower.1.desc
	picture = GFX_report_event_peace_deal
	
	is_triggered_only = yes
	fire_only_once = yes
	
	option = {
		name = RUS_fourthpower.1.a
		if = {
			limit = {
				country_exists = GEO
				country_exists = AZR
			}
			GEO = {
				econ_join_sphere_RUS = yes
			}
			ARM = {
				transfer_state = 230
				econ_join_sphere_RUS = yes
			}
			
			RUS = { puppet = ARM }
			RUS = { puppet = GEO }
		}
		else_if = {
			limit = {
				country_exists = ADY
				NOT = {
					country_exists = GEO
					country_exists = AZR
				}
			}
			ADY = {
				transfer_state = 230
			}
		}
		else = {
			RUS = {
				transfer_state = 230
			}
		}
	}
	option = {
		name = RUS_fourthpower.1.b
	}
}
